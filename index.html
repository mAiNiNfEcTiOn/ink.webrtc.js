WebRTC #1 - Default Markup

<video id="local" autoplay="autoplay" muted="muted"></video>
<video id="remote" autoplay="autoplay"></video>
<script>
Ink.requireModules(['Ink.Dom.Loaded_1','Ink.UI.WebRTC_1'],function(Loaded, WebRTC){
  Loaded.run(function(){
    window.webrtc = new WebRTC({
      localVideoElm: Ink.i('local'),
      remoteVideoElm: Ink.i('remote'),
      roomName: 'testInk1234'
    });
  });
});
</script>
